51 //calculate the mean obliquity of the ecliptic (in degrees)
52 static double calcMeanObliquityOfEcliptic(double t)
53 {
54  double seconds = 21.448 - t * (46.8150 + t * (0.00059 - t * 0.001813));
55  double e0 = 23.0 + (26.0 + (seconds / 60.0)) / 60.0;
56  return e0; // in degrees
57 }

59 //calculate the corrected obliquity of the ecliptic (in degrees)
60 static double calcObliquityCorrection(double t)
61 {
62  double e0 = calcMeanObliquityOfEcliptic(t);
63  double omega = 125.04 - 1934.136 * t;
64  double e = e0 + 0.00256 * cos(degToRad(omega));
65  return e; // in degrees
66 }

68 //calculate the Geometric Mean Longitude of the Sun (in degrees)
69 static double calcGeomMeanLongSun(double t)
70 {
71  double L0 = 280.46646 + t * (36000.76983 + 0.0003032 * t);
72  while (L0 > 360.0) {
73  L0 -= 360.0;
74  }
75  while (L0 < 0.0) {
76  L0 += 360.0;
77  }
78  return L0; // in degrees
79 }

81 //calculate the Geometric Mean Anomaly of the Sun (in degrees)
82 static double calcGeomMeanAnomalySun(double t)
83 {
84  double M = 357.52911 + t * (35999.05029 - 0.0001537 * t);
85  return M; // in degrees
86 }

88 //calculate the equation of center for the sun (in degrees)
89 static double calcSunEqOfCenter(double t)
90 {
91  double m = calcGeomMeanAnomalySun(t);
92  double mrad = degToRad(m);
93  double sinm = sin(mrad);
94  double sin2m = sin(mrad + mrad);
95  double sin3m = sin(mrad + mrad + mrad);
96  double C = sinm * (1.914602 - t * (0.004817 + 0.000014 * t)) +
97  sin2m * (0.019993 - 0.000101 * t) + sin3m * 0.000289;
98  return C; // in degrees
99 }

101 //calculate the true longitude of the sun (in degrees)
102 static double calcSunTrueLong(double t)
103 {
104  double l0 = calcGeomMeanLongSun(t);
105  double c = calcSunEqOfCenter(t);
106  double O = l0 + c;
107  return O; // in degrees
108 }

110 //calculate the apparent longitude of the sun (in degrees)
111 static double calcSunApparentLong(double t)
112 {
113  double o = calcSunTrueLong(t);
114  double omega = 125.04 - 1934.136 * t;
115  double lambda = o - 0.00569 - 0.00478 * sin(degToRad(omega));
116  return lambda; // in degrees
117 }

119 //calculate the declination of the sun (in degrees)
120 static double calcSunDeclination(double t)
121 {
122  double e = calcObliquityCorrection(t);
123  double lambda = calcSunApparentLong(t);
124 
125  double sint = sin(degToRad(e)) * sin(degToRad(lambda));
126  double theta = radToDeg(asin(sint));
127  return theta; // in degrees
128 }

130 //calculate the eccentricity of earth's orbit (unitless)
131 static double calcEccentricityEarthOrbit(double t)
132 {
133  double e = 0.016708634 - t * (0.000042037 + 0.0000001267 * t);
134  return e; // unitless
135 }

137 //calculate the difference between true solar time and mean solar time
138 //(output: equation of time in minutes of time)
139 static double calcEquationOfTime(double t)
140 {
141  double epsilon = calcObliquityCorrection(t);
142  double l0 = calcGeomMeanLongSun(t);
143  double e = calcEccentricityEarthOrbit(t);
144  double m = calcGeomMeanAnomalySun(t);
145 
146  double y = tan(degToRad(epsilon) / 2.0);
147  y *= y;
148 
149  double sin2l0 = sin(2.0 * degToRad(l0));
150  double sinm = sin(degToRad(m));
151  double cos2l0 = cos(2.0 * degToRad(l0));
152  double sin4l0 = sin(4.0 * degToRad(l0));
153  double sin2m = sin(2.0 * degToRad(m));
154 
155  double Etime = y * sin2l0 - 2.0 * e * sinm + 4.0 * e *
156  y * sinm * cos2l0 - 0.5 *
157  y * y * sin4l0 - 1.25 * e * e * sin2m;
158  return radToDeg(Etime) * 4.0; // in minutes of time
159 }

161 //calculate the hour angle of the sun at sunrise for the latitude (in radians)
162 static double calcHourAngleSunrise(double latitude, double solarDec)
163 {
164  double latRad = degToRad(latitude);
165  double sdRad = degToRad(solarDec);
166  double HAarg = (cos(degToRad(90.833)) /
167  (cos(latRad) * cos(sdRad)) - tan(latRad) * tan(sdRad));
168  double HA = acos(HAarg);
169  return HA; // in radians (for sunset, use -HA)
170 }

172 QTime KHolidays::SunRiseSet::utcSunrise(const QDate &date, double latitude, double longitude)
173 {
174  latitude = qMax(qMin(MaxLat, latitude), -MaxLat);
175  longitude = qMax(qMin(MaxLong, longitude), -MaxLong);
176 
177  double t = calcTimeJulianCent(date.toJulianDay());
178  double eqTime = calcEquationOfTime(t);
179  double solarDec = calcSunDeclination(t);
180  double hourAngle = calcHourAngleSunrise(latitude, solarDec);
181  double delta = longitude + radToDeg(hourAngle);
182  QTime timeUTC(0, 0);
183  if (std::isnan(delta)) {
184  return timeUTC;
185  }
186  timeUTC = timeUTC.addSecs((720 - (4.0 * delta) - eqTime) * 60);
187  return QTime(timeUTC.hour(),
188  timeUTC.second() > 29 ? timeUTC.minute() + 1 : timeUTC.minute(),
189  0);
190 }
